project(huffman)

include_directories(${CMAKE_SOURCE_DIR})

set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O2")

add_subdirectory(lib)
add_subdirectory(lib/test)
add_subdirectory(util)
